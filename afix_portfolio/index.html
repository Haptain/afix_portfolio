-- Payments Table
CREATE TABLE payments (
    payment_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    amount DECIMAL(10, 2),
    payment_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    payment_method VARCHAR(50),
    reference_number VARCHAR(100),
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);

-- Bandwidth Usage Table
CREATE TABLE usage_logs (
    log_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    date DATE,
    data_uploaded BIGINT,
    data_downloaded BIGINT,
    session_count INT,
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);

-- Monthly revenue per zone
SELECT zone, SUM(amount) AS monthly_revenue
FROM users u JOIN payments p ON u.user_id = p.user_id
WHERE MONTH(payment_date) = MONTH(CURDATE())
GROUP BY zone;

Java - Network Configuration Manager

import java.io.*;
import java.util.*;

public class NetworkConfigManager {
    private Map<String, Device> devices;

    public NetworkConfigManager() {
        this.devices = new HashMap<>();
    }

    public class Device {
        String ip, ssid, security;
        int bandwidth;

        Device(String ip, String ssid, String security, int bandwidth) {
            this.ip = ip;
            this.ssid = ssid;
            this.security = security;
            this.bandwidth = bandwidth;
        }
    }

    public void addDevice(String id, String ip, String ssid, String security, int bw) {
        devices.put(id, new Device(ip, ssid, security, bw));
        System.out.println("Device added: " + id);
    }

    public void configureDevice(String deviceId, String newSSID, String newSecurity) {
        Device d = devices.get(deviceId);
        if (d != null) {
            d.ssid = newSSID;
            d.security = newSecurity;
            System.out.println("Device " + deviceId + " configured");
        }
    }

    public void listAllDevices() {
        devices.forEach((id, device) -> {
            System.out.println(id + ": IP=" + device.ip + 
                             " SSID=" + device.ssid + 
                             " Security=" + device.security);
        });
    }

    public static void main(String[] args) {
        NetworkConfigManager manager = new NetworkConfigManager();
        manager.addDevice("K1", "192.168.100.1", "User_K1", "WPA2", 10);
        manager.addDevice("K2", "192.168.100.2", "User_K2", "WPA2", 10);
        manager.listAllDevices();
    }
}
Python - Bandwidth Monitor
Python
import requests
import json
from datetime import datetime

class BandwidthMonitor:
    def __init__(self, api_url):
        self.api_url = api_url
        self.usage_data = {}

    def fetch_device_stats(self, device_ip):
        """Get bandwidth stats from device"""
        try:
            response = requests.get(f"http://{device_ip}/api/stats")
            return response.json()
        except:
            return None

    def calculate_usage(self, uploaded_bytes, downloaded_bytes):
        """Convert bytes to GB"""
        total_bytes = uploaded_bytes + downloaded_bytes
        return total_bytes / (1024**3)

    def log_usage(self, user_id, usage_gb):
        """Log usage to database"""
        timestamp = datetime.now()
        self.usage_data[user_id] = {
            'usage_gb': usage_gb,
            'timestamp': timestamp.isoformat()
        }
        print(f"User {user_id}: {usage_gb:.2f} GB")

    def check_quota(self, user_id, limit_gb):
        """Check if user exceeded quota"""
        if user_id in self.usage_data:
            usage = self.usage_data[user_id]['usage_gb']
            return (limit_gb - usage) > 0
        return True

    def generate_report(self):
        """Generate usage report"""
        return json.dumps(self.usage_data, indent=2, default=str)

Android - Network Status Activity (Kotlin)
Kotlin
import android.os.Bundle
import androidx.appcompat.app.AppCompatActivity
import android.widget.TextView
import retrofit2.Retrofit
import retrofit2.http.GET

class NetworkStatusActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_network_status)

        val statusView = findViewById<TextView>(R.id.status_text)
        val speedView = findViewById<TextView>(R.id.speed_text)
        val usageView = findViewById<TextView>(R.id.usage_text)

        val api = Retrofit.Builder()
            .baseUrl("https://api.company.com/")
            .build()
            .create(NetworkAPI::class.java)

        val call = api.getStatus(userId = "user_001")
        call.enqueue(object : Callback<StatusResponse> {
            override fun onResponse(call: Call<StatusResponse>, 
                                   response: Response<StatusResponse>) {
                response.body()?.let {
                    statusView.text = "Status: ${it.status}"
                    speedView.text = "Speed: ${it.speed} Mbps"
                    usageView.text = "Usage: ${it.used_gb}/${it.limit_gb} GB"
                }
            }

            override fun onFailure(call: Call<StatusResponse>, t: Throwable) {
                statusView.text = "Error loading data"
            }
        })
    }
}

interface NetworkAPI {
    @GET("network/status")
    fun getStatus(@Query("user_id") userId: String): Call<StatusResponse>
}

data class StatusResponse(
    val status: String,
    val speed: Int,
    val used_gb: Float,
    val limit_gb: Int
)
